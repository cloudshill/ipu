<python>
class Plugin():
    count = 0

    @classmethod
    def on_plus(cls, data):
        cls.count += data
        return { 'count': cls.count }

    @classmethod
    def all(cls) -> dict:
        return {
            'plus': cls.on_plus
        }
</python>

class UserPlugin():

  def __init__(self):
      self.count = 0

  def on_plus(self, data):
      self.count += data
      return '<p> {} </p>'.format(self.count)

  def constructor(self):
      return '<p> {} </p>'.format(self.count)

  @classmethod
  def all(cls) -> dict:
      return {
          'plus': cls.on_plus
      }
